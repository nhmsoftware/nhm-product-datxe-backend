// docker-compose

services:
  # --- Service Backend ---
  app:
    build: .
    container_name: nhm-datxe-app
    restart: unless-stopped
    mem_limit: 200m
    volumes:
      - .:/var/www
    ports:
      - "8001:8000"
    env_file: .env
    environment:
      DB_HOST: host.docker.internal
      DB_PORT: 5432
      REDIS_HOST: host.docker.internal
      REDIS_PORT: 6379
    extra_hosts:
      - "host.docker.internal:host-gateway"

  # --- Service Realtime
  realtime:
    build:
      context: ../nhm-product-datxe-realtime
      dockerfile: Dockerfile
    container_name: nhm-realtime
    restart: unless-stopped
    volumes:
      - ../nhm-product-datxe-realtime:/app
      - /app/node_modules
    ports:
      - "8002:3001"
    env_file: ../nhm-product-datxe-realtime/.env
    environment:
      # Sửa từ 'redis' thành 'host.docker.internal' để dùng chung Redis với App
      REDIS_HOST: host.docker.internal
      REDIS_PORT: 6379
    # Thêm extra_hosts để Node.js biết host.docker.internal là máy Ubuntu của bạn
    extra_hosts:
      - "host.docker.internal:host-gateway"

networks:
  default:
    driver: bridge
    driver_opts:
      com.docker.network.driver.mtu: 1450


// Dockerfile

FROM php:8.3-cli-alpine

# Cài đúng những thứ cần thiết nhất
RUN apk add --no-cache \
    ca-certificates \
    curl \
    openssl \
    libpq \
    icu-libs \
    oniguruma \
    libzip

# Update cert
RUN update-ca-certificates

# Build PHP extensions (tối giản)
RUN apk add --no-cache --virtual .build-deps \
    $PHPIZE_DEPS \
    postgresql-dev \
    icu-dev \
    oniguruma-dev \
    libzip-dev \
    && docker-php-ext-install \
        pdo_pgsql \
        mbstring \
    && pecl install redis \
    && docker-php-ext-enable redis \
    && apk del .build-deps

WORKDIR /var/www

EXPOSE 8000

CMD ["php", "artisan", "serve", "--host=0.0.0.0", "--port=8000"]
